60/* Compute the "height" of a tree -- the number of
61 nodes along the longest path from the root node
62 down to the farthest leaf node.*/
63int height(node* node)
64{
65 if (node == NULL)
66 return 0;
67 else {
68 /* compute the height of each subtree */
69 int lheight = height(node->left);
70 int rheight = height(node->right);
71
72 /* use the larger one */
73 if (lheight > rheight) {
74 return (lheight + 1);
75 }
76 else {
77 return (rheight + 1);
78 }
79 }
